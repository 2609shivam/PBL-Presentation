Hybrid SQL Injection & XSS Detection System

A deterministic hybrid system that detects SQL Injection and Cross-Site Scripting attacks using static analysis, runtime monitoring, and signature-based detection — without machine learning.

Problem Statement

SQL Injection and XSS vulnerabilities persist due to improper input validation across web applications. Existing solutions either lack hybrid detection or rely heavily on machine learning, making them opaque, complex, and difficult to audit in production environments.

Literature Review

Existing tools like sqlmap focus on static penetration testing only. WAFs detect patterns but lack real-time proxy-level inspection. No open tool combines static analysis + live HTTP monitoring in a lightweight, rule-based pipeline.

Research Gap / Innovation

This system combines mitmproxy runtime interception, regex-based signature detection, and a scoring engine for both SQLi & XSS — deterministic, transparent, zero ML dependency.

System Methodology

Dataset / Input

HTTP requests (GET / POST / JSON) intercepted via mitmproxy, targeting a vulnerable Flask app on port 5000.

Architecture

addon.py → Scope gate → detector.py (SQLi scoring) + xss_detector.py → ui.py (Rich alert)

addon.py — mitmproxy entry point, routes all detections
from xss_detector import detect_xss from mitmproxy import http import json from detector import detect_payload, is_request_in_scope from ui import show_request, show_detection class SQLiAddon: def request(self, flow: http.HTTPFlow): # ── Scope gate: only inspect targeted endpoints ─────────── has_body = bool( flow.request.query or flow.request.urlencoded_form or "application/json" in flow.request.headers.get("Content-Type", "") ) if not is_request_in_scope(port=flow.request.port, path=flow.request.path, method=flow.request.method, has_body=has_body): return show_request(flow) # ── GET params: run both XSS + SQLi ─────────────────────── for k, v in flow.request.query.items(): xss_risk, xss_reason = detect_xss(v) sqli_risk, sqli_reason = detect_payload(v) if xss_risk: show_detection(flow, "GET", k, v, xss_reason, xss_risk, "XSS") if sqli_risk: show_detection(flow, "GET", k, v, sqli_reason, sqli_risk, "SQLi") # ── POST form data ───────────────────────────────────────── if flow.request.urlencoded_form: for k, v in flow.request.urlencoded_form.items(): xss_risk, xss_reason = detect_xss(v) sqli_risk, sqli_reason = detect_payload(v) if xss_risk: show_detection(flow, "POST", k, v, xss_reason, xss_risk, "XSS") if sqli_risk: show_detection(flow, "POST", k, v, sqli_reason, sqli_risk, "SQLi") addons = [SQLiAddon()]
detector.py — SQLi scoring + scope gate
import re # Target scope ───────────────────────────────────────────────── TARGET_PORTS = {5000} TARGET_PATH_PREFIXES = ("/login", "/search", "/register", "/api") ALLOWED_METHODS = {"GET", "POST"} # Strong regex patterns — immediate HIGH flag ─────────────────── STRONG_SQLI_PATTERNS = [ r"'\s*(or|and)\s+\d+=\d+", # ' OR 1=1 r"'\s*(or|and)\s+'?\w+'?='?\w+'?", # ' OR 'a'='a r"union\s+select", # UNION SELECT r"sleep\s*\(\s*\d+\s*\)", # time-based blind r"waitfor\s+delay", # MSSQL time-based ] def calculate_score(v: str) -> int: score = v.count("'") * 2 + v.count('"') * 2 score += sum(2 for k in SQL_KEYWORDS if re.search(rf"\b{k}\b", v)) if "--" in v or "/*" in v: score += 3 return score def detect_payload(value: str): v = value.lower().strip() for p in STRONG_SQLI_PATTERNS: # exact match → HIGH if re.search(p, v): return "HIGH", "Strong SQL injection pattern" score = calculate_score(v) # scoring fallback if score >= 5: return "HIGH", "High-confidence SQLi" elif score >= 3: return "MEDIUM", "Suspicious SQL-like structure" elif score >= 2: return "LOW", "Weak SQL indicators" return None, None
xss_detector.py — XSS regex engine
import re XSS_PATTERNS = [ r"<script.*?>.*?</script>", # classic <script> tag r"on\w+\s*=", # event handlers: onerror= onclick= … r"javascript:", # JS URI scheme r"<.*?on\w+=.*?>", # inline event inside any tag r"<img.*?src=.*?>", # img src injection ] def detect_xss(value: str): if not value: return None, None v = value.lower().strip() for pattern in XSS_PATTERNS: if re.search(pattern, v): return "HIGH", "XSS payload detected" # Weaker heuristic: raw angle brackets present if "<" in v and ">" in v: return "MEDIUM", "Possible HTML injection" return None, None
app.py — intentionally vulnerable Flask target application
from flask import Flask, render_template, request from markupsafe import escape, Markup import sqlite3 app = Flask(__name__) ALLOW_XSS = True # toggle XSS vulnerability ALLOW_SQLI = True # toggle SQLi vulnerability @app.route("/login", methods=["POST"]) def login(): username = request.form.get("username") password = request.form.get("password") conn = sqlite3.connect("database.db") if ALLOW_SQLI: # ❌ Vulnerable — raw string interpolation query = f"SELECT * FROM users WHERE username='{username}' AND password='{password}'" conn.cursor().execute(query) else: # ✅ Safe — parameterised query conn.cursor().execute("SELECT * FROM users WHERE username=? AND password=?", (username, password)) @app.route("/search") def search(): q = request.args.get("q") if ALLOW_XSS: output = f"<div>{Markup(q)}</div>" # ❌ Unescaped — XSS possible else: output = escape(q) # ✅ Safe if __name__ == "__main__": app.run(host="0.0.0.0", port=5000, debug=True)
ui.py — Rich console alert renderer
from rich.console import Console from rich.panel import Panel console = Console() def show_request(flow): console.print(f"[cyan]{flow.request.method}[/cyan] {flow.request.path}") def show_detection(flow, location, param, payload, reason, risk, attack_type="SQLi"): color = {"LOW": "yellow", "MEDIUM": "orange3", "HIGH": "red"}.get(risk, "white") body = f""" [bold]Method:[/bold] {flow.request.method} [bold]URL:[/bold] {flow.request.pretty_url} [bold]Location:[/bold] {location} [bold]Parameter:[/bold] {param} [bold]Payload:[/bold] {payload} [bold]Reason:[/bold] {reason} [bold]Risk:[/bold] {risk} """ console.print( Panel(body.strip(), title=f"🚨 {attack_type} Detected", border_style=color) )
